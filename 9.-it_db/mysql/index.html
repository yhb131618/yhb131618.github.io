__ Install #

  • yum install https://dev.mysql.com/get/mysql80-community-release-el7-6.noarch.rpm
  • yum install mysql-community-server -y
  • 설치 에러시
    • vi /etc/yum.repos.d/mysql-community.repo
    • gpgcheck=1 -»> gpgcheck=0
[mysql80-community]
name=MySQL 8.0 Community Server
baseurl=http://repo.mysql.com/yum/mysql-8.0-community/el/7/$basearch
enabled=1
gpgcheck=0
gpgkey=file:///etc/pki/rpm-gpg/RPM-GPG-KEY-mysql-2022

__ Replica #

__ __ Master Mysql #

  1. MySQL에 로그인합니다 (초기 임시 비밀번호 사용).

    mysql -u root -p

  2. 복제를 위한 별도의 사용자를 생성합니다.

CREATE USER 'replica'@'%' IDENTIFIED WITH mysql_native_password BY 'replica_password'; GRANT REPLICATION SLAVE ON *.* TO 'replica'@'%'; FLUSH PRIVILEGES;
  1. my.cnf 파일을 편집하여 복제 관련 설정을 추가합니다.
[mysqld] server-id=1 log_bin=mysql-bin bind-address=0.0.0.0`
  1. MySQL 서비스를 재시작합니다.
sudo systemctl restart mysqld
  1. 마스터 서버 확인
SHOW MASTER STATUS;
  1. 복제 설정(etc/my.cnf)
[mysqld]
server-id=1
log_bin=mysql-bin
binlog_do_db=example_db # 복제할 데이터베이스 이름. 필요에 따라 설정
bind-address = 0.0.0.0
  1. 재시작
sudo systemctl restart mysqld

__ __ Slave Mysql #

  1. 마스터 서버에서 다음 쿼리를 실행하여 로그 파일과 위치를 확인합니다. SHOW MASTER STATUS;

  2. 슬레이브 서버에서 다음 명령을 실행하여 복제를 설정합니다. MASTER_LOG_FILE과 MASTER_LOG_POS는 마스터 서버에서 확인한 값으로 대체합니다.

CHANGE MASTER TO 
MASTER_HOST='master_ip', 
MASTER_USER='replica', 
MASTER_PASSWORD='replica_password', 
MASTER_LOG_FILE='recorded_log_file_name', 
MASTER_LOG_POS=recorded_log_pos;  
START SLAVE;
  1. 복제 상태를 확인
SHOW SLAVE STATUS\G
SET GLOBAL read_only = OFF;
  1. 복제 설정(etc/my.cnf)
[mysqld]
server-id=2
log_bin=relay-bin
binlog_do_db=1
bind-address = 0.0.0.0
  1. 재시작
sudo systemctl restart mysqld

# /etc/keepalived/keepalived.conf 에서 기본 설정 예시
vrrp_script chk_mysql {
    script "killall -0 mysqld"       # MySQL이 실행 중인지 확인
    interval 2                      # 체크 간격 (초)
    weight 2                        # 스크립트 체크 결과에 따라 가중치 조정
}

vrrp_instance VI_1 {
    state MASTER                    # 초기 상태 설정 (MASTER 또는 BACKUP)
    interface eth0                  # VIP를 바인드할 네트워크 인터페이스
    virtual_router_id 51            # VRRP 라우터 ID (같은 VRRP 인스턴스에 대해 유니크해야 함)
    priority 100                    # 우선순위 설정 (높은 수가 우선)
    advert_int 1                    # 광고 간격 (초)
    authentication {
        auth_type PASS
        auth_pass 1111              # 인증 패스워드
    }
    virtual_ipaddress {
        192.168.1.10                # 가상 IP 주소
    }
    track_script {
        chk_mysql
    }
}

#!/bin/bash

# MySQL 서버 프로세스 (mysqld)의 PID를 체크합니다.
pid=$(pidof mysqld)

is_master=$(mysql -e "SHOW MASTER STATUS;" | grep "File") 


# PID가 존재하는지 여부를 확인합니다.
if [ -z "$pid" ]; then
    echo "MySQL 서버가 실행 중이지 않습니다."
    # MySQL 서버가 실행 중이지 않은 경우, 오류 코드로 종료합니다.
    # Keepalived에서 이 스크립트를 사용할 때, 이 오류 코드를 사용하여 failover를 트리거할 수 있습니다.
    exit 1
else
    echo "MySQL 서버가 실행 중입니다. PID: $pid"
    # MySQL 서버가 정상적으로 실행 중인 경우, 성공 코드로 종료합니다.
    if [ -z "$is_master" ]; then 
    # -z 문자열의 길이가 0인지 확인 한 후 0인면 true반환
    # 슬레이브서버이면 아래 코드 실행 
	   mysql -e "STOP SLAVE;"
	   mysql -e "RESET MASTER;" 
	fi
    exit 0
    
fi
